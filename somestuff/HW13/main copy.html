function createObstacles(){

     //obstacles

     fill(100,90,200);

     circle(cX,cY,50);

     if (cX > 400) {

         cX = 30;

     }

     else if (cX > 300) {

         cX += 20;

     }

     else if (cX <= 400) {

         cX += 10;

     } 

     

     if (cY > 300) {

         cY = 30;

     }

     else if (cY > 200) {

         cY += 20;

     } 

     else if (cY <= 300) {

         cY += 15;

     }


     fill(100,150,0);

     rect(rX,rY,20,20);

     if (rX>800){

         rX=0;

     }

     else if (rX<=800){

         rX+=20;

     }


     if (rY<1){

         rY =400;

     }

     else if (rY>1){

         rY-=5;

     }

}


function createBorders(A,B,C){

      /*walls

      A = 0

      B =800

      C =600

      */

      fill(A);

      rect(A,450,300,20);

      rect(800,450,-300,20);

      strokeWeight(20);

      line(A,A,B,A);

      line(A,A,A,C);

      line(B,A,B,C);

      line(A,C,B,C);

      //doors

      strokeWeight(12);

      //left door

      line(300,450,lLX,lLY);

      //right door

      line(rLX,rLY,500,450);

          rLX =400;

          rLY =450;

          lLX =390;

          lLY =450;

      }

}
